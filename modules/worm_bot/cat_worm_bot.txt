from modules.base_module import BaseModule

class PythonModule(BaseModule):
    def analyze_code(self, code: str) -> dict:
        """Analyzes Python code for issues and complexity (without radon)."""
        issues = []

        # Check for print statements
        if "print(" in code:
            issues.append("Avoid using 'print' statements. Use logging instead.")

        # Custom complexity analysis (basic heuristic)
        try:
            complexity = 0
            indent_levels = set()
            for line in code.split("\n"):
                stripped = line.lstrip()
                if stripped.startswith(("def ", "class ")):
                    complexity += 1
                indent_levels.add(len(line) - len(stripped))
            
            if len(indent_levels) > 5:  # Arbitrary threshold for complexity
                issues.append("High complexity detected due to deep indentation levels.")

        except Exception as e:
            issues.append(f"Complexity analysis failed: {str(e)}")

        return {"issues": issues}

    def fix_code(self, code: str) -> str:
        """Fixes Python code by replacing print statements with logging."""
        return code.replace("print(", "logging.info(")
